## 매수 종목 찾기

### 매수 대상 종목 검색

```python
from pykrx import stock
import datetime


MIN_CLOSE_PRICE = 2000
MAX_CLOSE_PRICE = 200000
MIN_VOLUME = 10000
MIN_VOLUME_PRICE = 20000000000
MIN_RATE = 0.0


prev_date = (datetime.datetime.now() - datetime.timedelta(1)).strftime("%Y%m%d")
target_date = datetime.datetime.now().strftime("%Y%m%d")
# prev_date = "20251110"
# target_date = "20251111"


df = stock.get_market_price_change(prev_date, prev_date)


# 종가, 거래량, 거래대금, 등락률 필터링
df = df[df["종가"] >= MIN_CLOSE_PRICE]
if MAX_CLOSE_PRICE > 0 and MAX_CLOSE_PRICE > MIN_CLOSE_PRICE:
    df = df[df["종가"] <= MAX_CLOSE_PRICE]
df = df[df["거래량"] >= MIN_VOLUME]
df = df[df["거래대금"] >= MIN_VOLUME_PRICE]
df = df[df["등락률"] > MIN_RATE]
print(f"조건에 맞는 {len(df)}개의 종목이 있습니다.")

prev_index = df.index
top_upward_df = df.sort_values(by=["등락률"], ascending=False)
top_upward_df = top_upward_df.to_csv(
    prev_date + "_top.csv", index=False, encoding="utf-8-sig"
)


# 다음날 모든 종목 데이터 가져오기
df = stock.get_market_price_change(target_date, target_date)
next_df = df.copy()
df = df.sort_values(by=["등락률"], ascending=False)
df = df.to_csv(target_date + "_top.csv", index=False, encoding="utf-8-sig")

# 이전날 조건에 맞는 종목만 필터링
df = next_df[next_df.index.isin(prev_index)]
df = df.sort_values(by=["등락률"], ascending=False)
df = df.to_csv(target_date + "_filtered_top.csv", index=False, encoding="utf-8-sig")
```
